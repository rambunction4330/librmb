114template <size_t NumModules>
115void SwerveDrive<NumModules>::driveCartesian(double xSpeed, double ySpeed,
116 double zRotation,
117 bool fieldOriented) {
118
119 Eigen::Vector2d robotRelativeVXY = Eigen::Vector2d(xSpeed, ySpeed);
120
121 if (fieldOriented) {
122 units::radian_t vXYRotationAngle = -gyro->getRotation().Radians();
123
124 Eigen::Matrix2d vXYRotation{
125 {std::cos(vXYRotationAngle()), -std::sin(vXYRotationAngle())},
126 {std::sin(vXYRotationAngle()), std::cos(vXYRotationAngle())}};
127
128 robotRelativeVXY = vXYRotation * robotRelativeVXY;
129 }
130
146 std::array<SwerveModulePower, NumModules> powers;
147 double largestPower = 1.0;
148
149 for (size_t i = 0; i < modules.size(); i++) {
150 SwerveModule &module = modules[i];
151
152 double output_x =
153 robotRelativeVXY.x() +
154 zRotation * module.getModuleTranslation().Y() / largestModuleDistance;
155
156 double output_y =
157 robotRelativeVXY.y() +
158 zRotation * -module.getModuleTranslation().X() / largestModuleDistance;
159
160 frc::Rotation2d moduleRotation{output_x, output_y};
161 double modulePower = std::sqrt(output_x * output_x + output_y * output_y);
162
163 powers[i] = SwerveModulePower{modulePower, moduleRotation};
164
165 if (modulePower > largestPower) {
166 largestPower = modulePower;
167 }
168 }
169
170 // Normalize
171 for (SwerveModulePower &power : powers) {
172 power.power /= largestPower;
173 }
174
175 // Optimize
176 for (size_t i = 0; i < modules.size(); i++) {
177 powers[i] =
178 SwerveModulePower::Optimize(powers[i], modules[i].getState().angle);
179 }
180
181 driveModulePowers(powers);
182}
